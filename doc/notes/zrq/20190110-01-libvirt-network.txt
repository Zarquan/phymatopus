# Generate the network addresses.
#[user@trop03]

    netsize=32
    netbase=200

    printmac()
        {
        local hibyte
        local lobyte
        local array
        local string
        hibyte=$((
            $1 / 256
            ))
        lobyte=$((
            $1 % 256
            ))
        array=('52' '54' '0' '0' '0' '0')
        array[4]=$(
            printf "%02X" ${hibyte}
            )
        array[5]=$(
            printf "%02X" ${lobyte}
            )
        string=${array[*]}
        echo ${string// /:}
        }

    printinet()
        {
        local hibyte
        local lobyte
        local array
        local string
        hibyte=$((
            $1 / 256
            ))
        lobyte=$((
            $1 % 256
            ))
        array=('192' '168' '0' '0')
        array[2]=$(
            printf "%d" ${hibyte}
            )
        array[3]=$(
            printf "%d" ${lobyte}
            )
        string=${array[*]}
        echo ${string// /.}
        }

    printnetwork()
        {
        local host=$1
        local type=$2
        local netn=$3

        base=$(((${netbase:?} * 256) + (${netn:?} * netsize)))

        local inet=$((base + 30))
        local cast=$((base + 31))

        echo "----"
        echo "${host} ${type}"
        echo "inet [$(printmac  ${inet})] [$(printinet ${inet})]"
        echo "cast [$(printinet ${cast})]"

        local virt
        for virt in {0..29}
        do
            temp=${host}[${virt}]
            name=${!temp}
            if [ ${type} == 'natted' ]
            then
                name=${name}-nat
            fi
            echo "virt $(printf "%16s" $name) [$(printf "%02X" ${virt})] [$(printmac  $((base + virt)))] [$(printinet $((base + virt)))]"
        done
        }

    for node in {1..8}
    do
        echo "---- ----"
        host=$(printf "trop%02x" ${node})
        echo "${host}"
        printnetwork \
            ${host} \
            'routed' \
            $(((${node} * 2) - 2))

        printnetwork \
            ${host} \
            'natted' \
            $(((${node} * 2) - 1))

    done
